#pragma once

#include <any>
#include <functional>
#include <unordered_map>
#include <vector>
#include <optional>
#include <memory>
#include <string>
#include <typeindex>
#include <shared_mutex>

namespace arxglue {

struct Context {
    std::any input;
    std::any output;
    // state теперь защищён мьютексом для потокобезопасного доступа
    std::unordered_map<std::string, std::any> state;
    mutable std::shared_mutex stateMutex;

    // Потокобезопасное получение значения (копия)
    template<typename T>
    T getState(const std::string& key) const {
        std::shared_lock lock(stateMutex);
        auto it = state.find(key);
        if (it != state.end()) {
            return std::any_cast<T>(it->second);
        }
        return T{};
    }

    // Потокобезопасная установка значения
    template<typename T>
    void setState(const std::string& key, T&& value) {
        std::unique_lock lock(stateMutex);
        state[key] = std::forward<T>(value);
    }

    // Проверка наличия ключа
    bool hasState(const std::string& key) const {
        std::shared_lock lock(stateMutex);
        return state.find(key) != state.end();
    }
};

using Component = std::function<void(Context&)>;

struct MultiConnection {
    Component source;
    std::vector<Component> targets;
    std::optional<std::function<void(Context&)>> transformer;
};

template<typename... Targets>
MultiConnection connect(Component source, Targets... targets) {
    return MultiConnection{source, {targets...}, std::nullopt};
}

template<typename... Targets>
MultiConnection connect(Component source, std::function<void(Context&)> transformer, Targets... targets) {
    return MultiConnection{source, {targets...}, transformer};
}

using NodeId = uint32_t;

struct PortInfo {
    std::string name;
    std::type_index type;
    bool required = true;
};

struct ComponentMetadata {
    std::string name;
    std::vector<PortInfo> inputs;
    std::vector<PortInfo> outputs;
    bool pure = false;
    bool volatile_ = false;
    bool threadSafe = false;
};

} // namespace arxglue

// Заголовки stb (реализация будет в отдельном .cpp)
#include "stb_image.h"
#include "stb_image_write.h"

namespace arxglue {

class Asset {
public:
    virtual ~Asset() = default;
    virtual std::type_index getType() const = 0;
    virtual void writeToFile(const std::string& path) const = 0;
};

class TextureAsset : public Asset {
public:
    TextureAsset(int w, int h, const std::vector<uint8_t>& data)
        : width(w), height(h), pixels(data) {}

    std::type_index getType() const override { return typeid(TextureAsset); }

    void writeToFile(const std::string& path) const override {
        std::string ext;
        size_t dotPos = path.find_last_of('.');
        if (dotPos != std::string::npos) {
            ext = path.substr(dotPos);
        }

        if (ext == ".png") {
            stbi_write_png(path.c_str(), width, height, 4, pixels.data(), width * 4);
        } else {
            std::ofstream f(path, std::ios::binary);
            int32_t w = width, h = height;
            f.write(reinterpret_cast<const char*>(&w), sizeof(w));
            f.write(reinterpret_cast<const char*>(&h), sizeof(h));
            uint32_t size = static_cast<uint32_t>(pixels.size());
            f.write(reinterpret_cast<const char*>(&size), sizeof(size));
            f.write(reinterpret_cast<const char*>(pixels.data()), pixels.size());
        }
    }

    static std::shared_ptr<TextureAsset> readFromFile(const std::string& path) {
        std::string ext;
        size_t dotPos = path.find_last_of('.');
        if (dotPos != std::string::npos) {
            ext = path.substr(dotPos);
        }

        if (ext == ".png") {
            int w, h, comp;
            unsigned char* data = stbi_load(path.c_str(), &w, &h, &comp, 4);
            if (!data) return nullptr;
            std::vector<uint8_t> pixels(data, data + w * h * 4);
            stbi_image_free(data);
            return std::make_shared<TextureAsset>(w, h, pixels);
        } else {
            std::ifstream f(path, std::ios::binary);
            if (!f) return nullptr;
            int32_t w, h;
            f.read(reinterpret_cast<char*>(&w), sizeof(w));
            f.read(reinterpret_cast<char*>(&h), sizeof(h));
            uint32_t size;
            f.read(reinterpret_cast<char*>(&size), sizeof(size));
            std::vector<uint8_t> pixels(size);
            f.read(reinterpret_cast<char*>(pixels.data()), size);
            return std::make_shared<TextureAsset>(w, h, pixels);
        }
    }

    int width, height;
    std::vector<uint8_t> pixels;
};

class MeshAsset : public Asset {
public:
    struct Vertex { float x, y, z; };
    std::vector<Vertex> vertices;
    std::vector<uint32_t> indices;

    MeshAsset(const std::vector<Vertex>& v, const std::vector<uint32_t>& i)
        : vertices(v), indices(i) {}

    std::type_index getType() const override { return typeid(MeshAsset); }

    void writeToFile(const std::string& path) const override {
        std::ofstream f(path, std::ios::binary);
        uint32_t vcount = static_cast<uint32_t>(vertices.size());
        f.write(reinterpret_cast<const char*>(&vcount), sizeof(vcount));
        f.write(reinterpret_cast<const char*>(vertices.data()), vcount * sizeof(Vertex));
        uint32_t icount = static_cast<uint32_t>(indices.size());
        f.write(reinterpret_cast<const char*>(&icount), sizeof(icount));
        f.write(reinterpret_cast<const char*>(indices.data()), icount * sizeof(uint32_t));
    }

    static std::shared_ptr<MeshAsset> readFromFile(const std::string& path) {
        std::ifstream f(path, std::ios::binary);
        if (!f) return nullptr;
        uint32_t vcount;
        f.read(reinterpret_cast<char*>(&vcount), sizeof(vcount));
        std::vector<Vertex> verts(vcount);
        f.read(reinterpret_cast<char*>(verts.data()), vcount * sizeof(Vertex));
        uint32_t icount;
        f.read(reinterpret_cast<char*>(&icount), sizeof(icount));
        std::vector<uint32_t> inds(icount);
        f.read(reinterpret_cast<char*>(inds.data()), icount * sizeof(uint32_t));
        return std::make_shared<MeshAsset>(verts, inds);
    }
};

class AssetManager {
public:
    static AssetManager& instance() {
        static AssetManager inst;
        return inst;
    }

    std::shared_ptr<TextureAsset> loadTexture(const std::string& path) {
        std::lock_guard<std::mutex> lock(m_mutex);
        auto it = m_cache.find(path);
        if (it != m_cache.end()) return std::dynamic_pointer_cast<TextureAsset>(it->second);
        auto tex = TextureAsset::readFromFile(path);
        if (tex) m_cache[path] = tex;
        return tex;
    }

    std::shared_ptr<MeshAsset> loadMesh(const std::string& path) {
        std::lock_guard<std::mutex> lock(m_mutex);
        auto it = m_cache.find(path);
        if (it != m_cache.end()) return std::dynamic_pointer_cast<MeshAsset>(it->second);
        auto mesh = MeshAsset::readFromFile(path);
        if (mesh) m_cache[path] = mesh;
        return mesh;
    }

    void saveAsset(const std::string& path, std::shared_ptr<Asset> asset) {
        std::lock_guard<std::mutex> lock(m_mutex);
        asset->writeToFile(path);
        m_cache[path] = asset;
    }

private:
    std::unordered_map<std::string, std::shared_ptr<Asset>> m_cache;
    std::mutex m_mutex;
};

} // namespace arxglue

// src/core/node.hpp
#pragma once
#include "../arxglue.hpp"
#include "../types/type_system.hpp"
#include <nlohmann/json.hpp>
#include <memory>
#include <vector>
#include <any>
#include <unordered_map>
#include <typeindex>
#include <iostream>
#include <mutex>
#include <shared_mutex>

namespace arxglue {

class INode {
public:
    virtual ~INode() = default;
    virtual void execute(Context& ctx) = 0;
    virtual ComponentMetadata getMetadata() const = 0;
    virtual void setParameter(const std::string& name, const std::any& value) = 0;
    virtual std::any getParameter(const std::string& name) const = 0;
    virtual void serialize(nlohmann::json& j) const = 0;
    virtual void deserialize(const nlohmann::json& j) = 0;

    void setId(NodeId id) { m_id = id; }
    NodeId getId() const { return m_id; }

    bool isDirty() const { return m_dirty; }
    void setDirty(bool dirty) { m_dirty = dirty; }

    void setCachedOutput(const std::any& output) { m_cachedOutput = output; }
    std::any getCachedOutput() const { return m_cachedOutput; }

    // Получить значение входа с автоматической конвертацией к ожидаемому типу
    std::any getInputValue(Context& ctx, int portIndex) const {
        const auto& inputs = getMetadata().inputs;
        if (portIndex < 0 || portIndex >= static_cast<int>(inputs.size())) {
            return {};
        }
        std::string key = "in_" + std::to_string(m_id) + "_" + std::to_string(portIndex);
        std::shared_lock stateLock(ctx.stateMutex);   // переименовано
        auto it = ctx.state.find(key);
        if (it == ctx.state.end()) {
            return {};
        }
        const std::any& rawValue = it->second;
        const std::type_index& expectedType = inputs[portIndex].type;
        if (std::type_index(rawValue.type()) == expectedType) {
            return rawValue;
        }
        try {
            return TypeSystem::instance().convertTo(rawValue, expectedType);
        } catch (const std::exception& e) {
            std::cerr << "Type conversion error in node " << m_id << " port " << portIndex << ": " << e.what() << std::endl;
            return {};
        }
    }

    void setOutputValue(Context& ctx, int portIndex, const std::any& value) {
        const auto& outputs = getMetadata().outputs;
        if (portIndex >= 0 && portIndex < static_cast<int>(outputs.size())) {
            std::string key = "out_" + std::to_string(m_id) + "_" + std::to_string(portIndex);
            std::unique_lock lock(ctx.stateMutex);
            ctx.state[key] = value;
        }
        ctx.output = value;
    }

    // Проверка, изменились ли входы по сравнению с последним выполнением
    bool areInputsChanged(const Context& ctx) {
        const auto& inputs = getMetadata().inputs;
        if (inputs.empty()) return false;

        std::lock_guard<std::mutex> inputLock(m_inputMutex);
        if (m_lastInputs.size() != inputs.size()) {
            m_lastInputs.resize(inputs.size());
            return true;
        }

        for (size_t i = 0; i < inputs.size(); ++i) {
            std::string key = "in_" + std::to_string(m_id) + "_" + std::to_string(i);
            std::any currentValue;
            {
                std::shared_lock lock(ctx.stateMutex);
                auto it = ctx.state.find(key);
                if (it != ctx.state.end()) {
                    currentValue = it->second;
                }
            }
            if (!TypeSystem::compareAny(currentValue, m_lastInputs[i])) {
                m_lastInputs[i] = currentValue;
                return true;
            }
        }
        return false;
    }

    // Обновить сохранённые значения входов после выполнения
    void updateLastInputs(const Context& ctx) {
        const auto& inputs = getMetadata().inputs;
        std::lock_guard<std::mutex> inputLock(m_inputMutex);   
        m_lastInputs.resize(inputs.size());
        for (size_t i = 0; i < inputs.size(); ++i) {
            std::string key = "in_" + std::to_string(m_id) + "_" + std::to_string(i);
            std::shared_lock stateLock(ctx.stateMutex);        
            auto it = ctx.state.find(key);
            m_lastInputs[i] = (it != ctx.state.end()) ? it->second : std::any{};
        }
    }

    std::type_index getInputPortType(int portIndex) const {
        const auto& inputs = getMetadata().inputs;
        if (portIndex >= 0 && portIndex < static_cast<int>(inputs.size())) {
            return inputs[portIndex].type;
        }
        return typeid(void);
    }

    std::type_index getOutputPortType(int portIndex) const {
        const auto& outputs = getMetadata().outputs;
        if (portIndex >= 0 && portIndex < static_cast<int>(outputs.size())) {
            return outputs[portIndex].type;
        }
        return typeid(void);
    }

    bool areParametersChanged() const {
        size_t currentHash = computeParamsHash();
        if (currentHash != m_lastParamsHash) {
            m_lastParamsHash = currentHash;
            return true;
        }
        return false;
    }

    void clearLastInputs() {
        std::lock_guard<std::mutex> inputLock(m_inputMutex);
        m_lastInputs.clear();
    }

protected:
    virtual size_t computeParamsHash() const { return 0; }

private:
    NodeId m_id = 0;
    bool m_dirty = true;
    std::any m_cachedOutput;
    mutable std::vector<std::any> m_lastInputs;
    mutable std::mutex m_inputMutex;
    mutable size_t m_lastParamsHash = 0;
};

} // namespace arxglue

#include "../assets/asset_manager.hpp"
#include <memory>
#include <string>

namespace arxglue {

class AssetNode : public INode {
public:
    AssetNode() = default;
    explicit AssetNode(const std::string& assetPath, const std::string& assetType = "texture")
        : m_assetPath(assetPath), m_assetType(assetType) {}

    void execute(Context& ctx) override;

    ComponentMetadata getMetadata() const override {
        std::type_index outputType = typeid(void);
        if (m_assetType == "texture") {
            outputType = typeid(std::shared_ptr<TextureAsset>);
        } else if (m_assetType == "mesh") {
            outputType = typeid(std::shared_ptr<MeshAsset>);
        }

        return {"Asset",
                {},
                {{"asset", outputType}},
                true,   // pure
                false}; // not volatile
    }

    void setParameter(const std::string& name, const std::any& value) override {
        if (name == "path") m_assetPath = std::any_cast<std::string>(value);
        else if (name == "type") m_assetType = std::any_cast<std::string>(value);
    }

    std::any getParameter(const std::string& name) const override {
        if (name == "path") return m_assetPath;
        if (name == "type") return m_assetType;
        return {};
    }

    void serialize(nlohmann::json& j) const override {
        j["type"] = "AssetNode";
        j["params"]["path"] = m_assetPath;
        j["params"]["assetType"] = m_assetType;
    }

    void deserialize(const nlohmann::json& j) override {
        if (j.contains("params")) {
            if (j["params"].contains("path")) m_assetPath = j["params"]["path"].get<std::string>();
            if (j["params"].contains("assetType")) m_assetType = j["params"]["assetType"].get<std::string>();
        }
    }

private:
    std::string m_assetPath;
    std::string m_assetType = "texture";
    std::shared_ptr<Asset> m_cachedAsset; // хранится как базовый тип
};

} // namespace arxglue

template<typename T>
class ConstantNode : public INode {
public:
    ConstantNode(T value) : m_value(value) {}

    void execute(Context& ctx) override {
        setOutputValue(ctx, 0, m_value);
    }

    ComponentMetadata getMetadata() const override {
        return {"Constant" + std::string(typeid(T).name()),
                {}, // нет входов
                {{"out", typeid(T)}},
                true, false};
    }

    void setParameter(const std::string& name, const std::any& value) override {
        if (name == "value") {
            m_value = std::any_cast<T>(value);
            setDirty(true); // <-- добавить
        }
    }

    std::any getParameter(const std::string& name) const override {
        if (name == "value") return m_value;
        return {};
    }

    void serialize(nlohmann::json& j) const override {
        if constexpr (std::is_same_v<T, int>) {
            j["type"] = "ConstantInt";
        } else if constexpr (std::is_same_v<T, float>) {
            j["type"] = "ConstantFloat";
        } else {
            j["type"] = "ConstantUnknown";
        }
        j["params"]["value"] = m_value;
    }

    void deserialize(const nlohmann::json& j) override {
        if (j.contains("params") && j["params"].contains("value")) {
            m_value = j["params"]["value"].get<T>();
        }
    }

protected:
    size_t computeParamsHash() const override {
        return std::hash<T>{}(m_value);
    }

private:
    T m_value;
};

class SlowNode : public INode {
public:
    SlowNode() : m_delayMs(100) {}
    explicit SlowNode(int delayMs) : m_delayMs(delayMs) {}

    void execute(Context& ctx) override {
        // Искусственная задержка для демонстрации параллелизма
        std::this_thread::sleep_for(std::chrono::milliseconds(m_delayMs));

        // Ожидаем на входе целое число, на выходе возвращаем его же
        int val = std::any_cast<int>(getInputValue(ctx, 0));
        setOutputValue(ctx, 0, val);
    }

    ComponentMetadata getMetadata() const override {
        return {
            "SlowNode",
            {{"in", typeid(int), true}},
            {{"out", typeid(int)}},
            true,   // pure (результат зависит только от входа)
            false   // не volatile
        };
    }

    void setParameter(const std::string& name, const std::any& value) override {
        if (name == "delay") {
            m_delayMs = std::any_cast<int>(value);
        }
    }

    std::any getParameter(const std::string& name) const override {
        if (name == "delay") return m_delayMs;
        return {};
    }

    void serialize(nlohmann::json& j) const override {
        j["type"] = "SlowNode";
        j["params"]["delay"] = m_delayMs;
    }

    void deserialize(const nlohmann::json& j) override {
        if (j.contains("params") && j["params"].contains("delay")) {
            m_delayMs = j["params"]["delay"].get<int>();
        }
    }

private:
    int m_delayMs;
};

#pragma once
#include "../core/node.hpp"
#include <memory>
#include <unordered_map>
#include <functional>

namespace arxglue {

class NodeFactory {
public:
    using Creator = std::function<std::unique_ptr<INode>()>;

    static NodeFactory& instance() {
        static NodeFactory inst;
        return inst;
    }

    void registerNode(const std::string& type, Creator creator) {
        m_creators[type] = std::move(creator);
    }

    std::unique_ptr<INode> createNode(const std::string& type) {
        auto it = m_creators.find(type);
        if (it == m_creators.end()) return nullptr;
        return it->second();
    }

private:
    std::unordered_map<std::string, Creator> m_creators;
};

} // namespace arxglue

#pragma once

#include <vector>
#include <queue>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <functional>
#include <future>
#include <atomic>

namespace arxglue {

class ThreadPool {
public:
    explicit ThreadPool(size_t numThreads = std::thread::hardware_concurrency()) {
        for (size_t i = 0; i < numThreads; ++i) {
            m_workers.emplace_back([this] {
                while (true) {
                    std::function<void()> task;
                    {
                        std::unique_lock<std::mutex> lock(m_mutex);
                        m_cv.wait(lock, [this] { return m_stop || !m_tasks.empty(); });
                        if (m_stop && m_tasks.empty()) return;
                        task = std::move(m_tasks.front());
                        m_tasks.pop();
                    }
                    task();
                }
            });
        }
    }

    ~ThreadPool() {
        {
            std::unique_lock<std::mutex> lock(m_mutex);
            m_stop = true;
        }
        m_cv.notify_all();
        for (std::thread& worker : m_workers) {
            if (worker.joinable()) worker.join();
        }
    }

    template<typename F, typename... Args>
    auto enqueue(F&& f, Args&&... args) -> std::future<typename std::invoke_result_t<F, Args...>> {
        using return_type = typename std::invoke_result_t<F, Args...>;
        auto task = std::make_shared<std::packaged_task<return_type()>>(
            std::bind(std::forward<F>(f), std::forward<Args>(args)...)
        );
        std::future<return_type> res = task->get_future();
        {
            std::unique_lock<std::mutex> lock(m_mutex);
            if (m_stop) throw std::runtime_error("enqueue on stopped ThreadPool");
            m_tasks.emplace([task]() { (*task)(); });
        }
        m_cv.notify_one();
        return res;
    }

private:
    std::vector<std::thread> m_workers;
    std::queue<std::function<void()>> m_tasks;
    std::mutex m_mutex;
    std::condition_variable m_cv;
    std::atomic<bool> m_stop{false};
};

} // namespace arxglue

// src/types/type_system.hpp
#pragma once
#include <any>
#include <typeindex>
#include <functional>
#include <unordered_map>
#include <stdexcept>
#include <iostream>
#include <memory>

namespace arxglue::render {
    class Texture;
    class Mesh;
    class Material;
    class Scene;
    class Camera;
}

namespace arxglue {

enum class TypeId : uint32_t {
    Unknown = 0,
    Int,
    Float,
    String,
    Texture,
    Mesh,
    Material,
    Scene,
    Camera
};

struct PairHash {
    template<typename T1, typename T2>
    std::size_t operator()(const std::pair<T1, T2>& p) const {
        auto h1 = std::hash<T1>{}(p.first);
        auto h2 = std::hash<T2>{}(p.second);
        return h1 ^ (h2 << 1);
    }
};

class TypeSystem {
public:
    static TypeSystem& instance() {
        static TypeSystem inst;
        return inst;
    }

    void registerType(TypeId id, const std::type_index& stdType, TypeId parent = TypeId::Unknown) {
        m_typeInfo.emplace(id, TypeInfo(stdType, parent));
        m_typeToId[stdType] = id;
    }

    TypeId getTypeId(const std::type_index& type) const {
        auto it = m_typeToId.find(type);
        if (it != m_typeToId.end()) return it->second;
        return TypeId::Unknown;
    }

    bool canConvert(TypeId from, TypeId to) const {
        if (from == to) return true;
        auto it = m_typeInfo.find(from);
        if (it == m_typeInfo.end()) return false;
        TypeId cur = it->second.parent;
        while (cur != TypeId::Unknown) {
            if (cur == to) return true;
            auto pit = m_typeInfo.find(cur);
            if (pit == m_typeInfo.end()) break;
            cur = pit->second.parent;
        }
        return m_converters.find({from, to}) != m_converters.end();
    }

    std::any convert(const std::any& value, TypeId from, TypeId to) const {
        if (from == to) return value;
        auto key = std::make_pair(from, to);
        auto it = m_converters.find(key);
        if (it != m_converters.end()) {
            return it->second(value);
        }
        auto infoIt = m_typeInfo.find(from);
        if (infoIt != m_typeInfo.end() && infoIt->second.parent != TypeId::Unknown) {
            return convert(value, infoIt->second.parent, to);
        }
        throw std::runtime_error("No conversion available from type " + std::to_string(static_cast<int>(from)) +
                                 " to " + std::to_string(static_cast<int>(to)));
    }

    std::any convertTo(const std::any& value, const std::type_index& targetType) const {
        TypeId fromId = getTypeId(value.type());
        TypeId toId = getTypeId(targetType);
        if (fromId == TypeId::Unknown || toId == TypeId::Unknown) {
            throw std::runtime_error("Unknown type(s) in conversion");
        }
        return convert(value, fromId, toId);
    }

    void registerConverter(TypeId from, TypeId to, std::function<std::any(const std::any&)> conv) {
        m_converters[{from, to}] = std::move(conv);
    }

    std::type_index getStdTypeIndex(TypeId id) const {
        auto it = m_typeInfo.find(id);
        if (it != m_typeInfo.end()) return it->second.stdType;
        return typeid(void);
    }

    static bool compareAny(const std::any& a, const std::any& b) {
        if (!a.has_value() && !b.has_value()) return true;
        if (!a.has_value() || !b.has_value()) return false;
        if (std::type_index(a.type()) != std::type_index(b.type())) return false;
        const std::type_info& ti = a.type();
        if (ti == typeid(int)) {
            return std::any_cast<int>(a) == std::any_cast<int>(b);
        } else if (ti == typeid(float)) {
            return std::any_cast<float>(a) == std::any_cast<float>(b);
        } else if (ti == typeid(std::string)) {
            return std::any_cast<std::string>(a) == std::any_cast<std::string>(b);
        } else if (ti == typeid(bool)) {
            return std::any_cast<bool>(a) == std::any_cast<bool>(b);
        }
        return false; // для сложных типов считаем разными
    }



private:
    struct TypeInfo {
        std::type_index stdType;
        TypeId parent;
        TypeInfo(const std::type_index& t, TypeId p) : stdType(t), parent(p) {}
    };

    std::unordered_map<TypeId, TypeInfo> m_typeInfo;
    std::unordered_map<std::type_index, TypeId> m_typeToId;
    std::unordered_map<std::pair<TypeId, TypeId>, std::function<std::any(const std::any&)>, PairHash> m_converters;
};

inline void initBasicTypes() {
    auto& ts = TypeSystem::instance();
    ts.registerType(TypeId::Int, typeid(int));
    ts.registerType(TypeId::Float, typeid(float));
    ts.registerType(TypeId::String, typeid(std::string));
    ts.registerType(TypeId::Texture, typeid(std::shared_ptr<class TextureAsset>));
    ts.registerType(TypeId::Mesh, typeid(std::shared_ptr<class MeshAsset>));
    ts.registerType(TypeId::Texture, typeid(std::shared_ptr<render::Texture>));
    ts.registerType(TypeId::Mesh, typeid(std::shared_ptr<render::Mesh>));
    ts.registerType(TypeId::Material, typeid(std::shared_ptr<render::Material>));
    ts.registerType(TypeId::Scene, typeid(std::shared_ptr<render::Scene>));
    ts.registerType(TypeId::Camera, typeid(std::shared_ptr<render::Camera>));

    ts.registerConverter(TypeId::Int, TypeId::Float, [](const std::any& v) -> std::any {
        return static_cast<float>(std::any_cast<int>(v));
    });
    ts.registerConverter(TypeId::Float, TypeId::Int, [](const std::any& v) -> std::any {
        return static_cast<int>(std::any_cast<float>(v));
    });
}

3Tone Engine – Документация проекта
Введение
3Tone — это амбициозный проект по созданию игрового движка и среды разработки, вдохновлённый архитектурой Werkkzeug4 (инструментарий демосцены) и декларативным протоколом композиции компонентов ArxGlue. Основная идея: объединить процедурную генерацию контента (узлы-генераторы, граф вычислений) с традиционными ассетами (модели, текстуры, анимации) в единую гибкую парадигму, где разработчик может на лету смешивать префабы и процедурные данные, а также «запекать» результаты в ассеты для финального билда.
Ключевые принципы:
Декларативность – граф узлов описывает, что нужно вычислить, а не как.
Типизированные соединения – каждый узел имеет определённый тип входов/выходов, система сама вставляет конвертеры.
Контекст выполнения – единое хранилище для входных данных, результатов и состояния (префабы, ресурсы, глобальные настройки).
Гибридный пайплайн – ассеты и генераторы живут в одном графе, поддерживается «запечка» подграфов в ассеты.
Кроссплатформенность – код на C++ 20, с возможностью использования Vulkan/Metal/DirectX 12 через абстракцию.
Цель документа – зафиксировать архитектурные решения, протоколы и планы разработки, чтобы превратить разрозненные идеи в работающий движок.

Заключение и ссылки

1. Архитектура ядра
1.1. Компоненты (узлы)
Компонент (или узел) — это базовая единица вычислений в движке 3Tone. Компонент получает входные данные, выполняет некоторую операцию и возвращает выходные данные.
В C++ компонент представлен типом Component:
cpp
using Component = std::function<void(Context&)>;
Любая сущность, которая может быть вызвана с контекстом (функция, лямбда, объект с operator()), может выступать компонентом. Это обеспечивает максимальную гибкость: компонентами являются как элементарные операции (сложение, умножение), так и сложные генераторы (шум Перлина, рендер сцены).
Варианты компонентов:
Примитивные – не имеют входов (только контекст с начальным input или без него).
Вычислительные – имеют один или несколько входов (через контекст или будущие порты).
Контейнерные – хранят ассеты (текстуры, меши) и могут выступать как источники данных.
Каждый компонент должен быть чистой функцией относительно входных данных (если не используется состояние). Однако контекст позволяет хранить изменяемое состояние для узлов, которым оно необходимо (например, генераторы случайных чисел, таймеры).
1.2. Контекст выполнения
Context — это единое хранилище данных, передаваемое через все компоненты. Он содержит:
input — входное значение (тип std::any).
output — выходное значение.
state — словарь для произвольного состояния (префабы, ресурсы, кэши).
cpp
struct Context {
    std::any input;
    std::any output;
    std::unordered_map<std::string, std::any> state;
};
Контекст решает две задачи:
Передача данных между компонентами в цепочке.
Предоставление доступа к глобальным или долгоживущим объектам (загруженные текстуры, менеджер ресурсов).
Почему не явные порты?
В духе минимализма и гибкости протокола ArxGlue, мы не требуем от пользователя декларировать порты каждого узла. Вместо этого компоненты читают и пишут input/output. Это упрощает создание простых цепочек. Для сложных графов с несколькими входами можно использовать state или ввести более строгую систему портов (это будет расширением, опциональным).
1.3. Соединения и ветвление
Соединение (Connection) связывает один источник с одной или несколькими целями. Также может включать опциональный трансформер — функцию, которая преобразует выход источника перед подачей в цели.
cpp
struct MultiConnection {
    Component source;
    std::vector<Component> targets;
    std::optional<std::function<void(Context&)>> transformer;
};
Функция connect позволяет создавать соединения с произвольным числом целей:
cpp
auto conn = connect(source, target1, target2, target3);
// или с трансформером
auto conn2 = connect(source, target1, target2, transformer);
При выполнении соединения:
Вызывается source(ctx). Результат сохраняется.
Если есть трансформер, он вызывается с тем же контекстом (преобразует output).
Для каждой цели создаётся копия контекста (или отдельный контекст), в input помещается результат источника, и цель выполняется.
После обработки всех целей output исходного контекста восстанавливается (если требуется для следующего соединения).
Ветвление поддерживается «из коробки»: один источник может питать множество целей. Это соответствует потребностям процедурной генерации, где результат одного узла используется в нескольких местах.
1.4. Граф и исполнитель
Граф — это набор соединений. В простейшей форме граф представляет собой последовательность соединений, выполняемых одно за другим. Однако для поддержки произвольных зависимостей (когда выход одного узла идёт на вход узла, расположенного выше в списке) требуется топологическая сортировка.
Базовый исполнитель (Graph::execute) пока линейный, но архитектура позволяет в дальнейшем ввести:
Топологическую сортировку узлов.
Вычисление только «грязных» узлов (чей вход изменился).
Многопоточное выполнение независимых ветвей.
План развития исполнителя:
v0.1 – последовательное выполнение соединений (как в текущей реализации).
v0.2 – построение ориентированного ациклического графа (ОАГ) из соединений, топологическая сортировка.
v0.3 – кэширование результатов узлов и инкрементальное обновление.
v1.0 – многопоточный планировщик с пулом потоков.
1.5. Система типов и конвертации
В текущей реализации типы данных представлены std::any. Это удобно для прототипирования, но не даёт контроля над совместимостью. Для серьёзного движка необходима система типов, вдохновлённая wType из Werkkzeug4.
Планируемая система типов:
Каждый тип имеет уникальный идентификатор (enum или std::type_index).
Поддерживается иерархия (наследование типов) – например, Texture2D является подтипом Texture.
Для каждого типа определены возможные конвертации (например, float -> int через округление, Texture -> RenderTarget).
При соединении двух узлов, если тип выхода источника не совпадает с ожидаемым типом входа цели, исполнитель автоматически вставляет узел-конвертер (или вызывает встроенную функцию).
Реализация (эскиз):
cpp
enum class TypeId { Int, Float, Vec2, Texture, Mesh, ... };

struct TypeInfo {
    TypeId id;
    TypeInfo* parent;          // для наследования
    std::function<bool(const std::any&)> validator;
    std::function<std::any(const std::any&)> converterFrom[TypeIdCount];
};
Это позволит сделать граф типобезопасным и избавит пользователя от ручного приведения типов.
1.6. Кэширование и «ленивые» вычисления
Кэш – механизм, сохраняющий результат выполнения узла до тех пор, пока его входы не изменятся. Это критично для производительности в сложных графах, где один и тот же узел используется многократно.
Правила кэширования:
Узел считается «чистым», если его выход детерминированно зависит от входов и состояния.
Результат узла сохраняется после выполнения.
При изменении любого входа (или глобального состояния, на которое узел подписан) кэш инвалидируется.
Пользователь может пометить узел как Volatile (например, генератор случайных чисел, чтение времени), чтобы отключить кэширование.
Интеграция с графом:
Исполнитель перед вызовом узла проверяет, есть ли валидный кэш.
Если есть и входы не менялись – возвращает кэшированное значение, не вызывая узел.
При сериализации графа кэш не сохраняется (только опционально для «запеченных» узлов).
LRU-политика для ограничения памяти (как в Werkkzeug4) может быть добавлена на поздних этапах.


2. Протокол ArxGlue (C++ реализация)
Протокол ArxGlue — это минималистичный декларативный язык композиции компонентов, изначально реализованный на Python. В контексте движка 3Tone он служит ядром для построения графов вычислений, обеспечивая гибкое соединение узлов и передачу данных через контекст.
Данный раздел описывает текущую C++ реализацию протокола, её ключевые особенности, ограничения и планы по расширению.
2.1. Контекст (Context)
Context — единственный канал передачи данных между компонентами. Он содержит три поля:
cpp
struct Context {
    std::any input;                           // входное значение
    std::any output;                          // выходное значение
    std::unordered_map<std::string, std::any> state; // состояние/ресурсы
};
input / output — основные потоки данных. Компонент читает input и записывает output. После выполнения компонента output становится input для следующего (в линейной цепочке) или для каждой цели (при ветвлении).
state — универсальное хранилище для:
Глобальных настроек (разрешение экрана, время, delta time).
Ресурсов (загруженные текстуры, меши, звуки).
Кэшей и промежуточных результатов.
Пользовательских данных, которые необходимо передать между компонентами вне основного потока.
Особенности:
std::any обеспечивает типобезопасность в рантайме (проверка через any_cast). Это удобно для прототипирования, но требует явного приведения.
Контекст копируется при ветвлении (каждая цель получает собственную копию). Это позволяет целям изменять своё состояние, не влияя на другие ветви. В будущем можно ввести семантику перемещения для оптимизации.
2.2. Компонент как std::function
Компонент — это любой вызываемый объект с сигнатурой void(Context&). В C++ это выражается через std::function:
cpp
using Component = std::function<void(Context&)>;
Что может быть компонентом:
Обычная функция: void myComp(Context& ctx) { ... }
Лямбда: auto comp = [](Context& ctx) { ... };
Объект класса с перегруженным operator(): struct Add { void operator()(Context& ctx) const { ... } };
Результат std::bind или другой std::function.
Преимущества:
Гибкость – можно создавать компоненты «на лету».
Совместимость со стандартной библиотекой.
Возможность хранить компоненты в контейнерах.
Недостатки:
std::function имеет небольшие накладные расходы (виртуальный вызов). Для высокопроизводительных участков можно использовать шаблонные компоненты, но в ядре графа это обычно не критично.
2.3. Соединения с несколькими целями
В отличие от классического Dataflow, где соединение соединяет один выход с одним входом, в протоколе ArxGlue соединение может иметь один источник и произвольное количество целей. Это реализует ветвление без необходимости в специальных узлах-разветвителях.
cpp
struct MultiConnection {
    Component source;
    std::vector<Component> targets;
    std::optional<std::function<void(Context&)>> transformer;
};
Создание соединения:
cpp
// без трансформера
auto conn = connect(source, target1, target2, target3);
// с трансформером (последний аргумент – функция)
auto conn = connect(source, target1, target2, transformer);
Семантика выполнения:
Вызвать source(ctx). Сохранить результат.
Если есть transformer, вызвать его с тем же контекстом (он может изменить ctx.output).
Для каждой цели:
Скопировать контекст (или создать новый).
Установить branchCtx.input = ctx.output (результат источника/трансформера).
Вызвать target(branchCtx).
(Опционально) объединить выходы целей, но по умолчанию игнорируются.
Восстановить ctx.output для следующего соединения (если необходимо).
Почему копирование контекста?
Чтобы цели не затирали данные друг друга и не мешали последующим соединениям. Если цели не изменяют состояние (чистые функции), копирование можно оптимизировать, но для простоты и надёжности оставляем копирование.
2.4. Базовый граф последовательного выполнения
Граф — это просто список соединений, выполняемых в порядке добавления.
cpp
class Graph {
    std::vector<MultiConnection> connections_;
public:
    void addConnection(const MultiConnection& conn);
    void execute(Context& ctx);
};
Выполнение:
Последовательно проходим по всем соединениям.
Каждое соединение выполняется в соответствии с описанной выше семантикой.
Выход последнего соединения не используется (если только граф не является частью большего графа).
Ограничения:
Граф линейный – нет возможности направить выход одного соединения на вход другого, если этот другой расположен выше в списке (обратная связь). Для произвольных зависимостей необходима топологическая сортировка.
Нет поддержки циклов (хотя их можно эмулировать через состояние и явные переходы).
Кэширование отсутствует – каждый вызов execute пересчитывает все компоненты.
Планы расширения:
Режим сборки графа с автоматическим определением порядка по зависимостям (через анализ, какие компоненты читают/пишут в state или через явное объявление портов).
Инкрементальное обновление – перевычислять только компоненты, у которых изменились входы.
Многопоточное выполнение – независимые соединения (или цели в одном соединении) могут выполняться параллельно.
2.5. Пример использования (из предыдущего раздела)
cpp
void generateNumber(Context& ctx) { ctx.output = 42; }
void multiplyByTwo(Context& ctx) { int v = any_cast<int>(ctx.input); ctx.output = v * 2; }
void print(Context& ctx) { cout << any_cast<int>(ctx.output) << endl; }

Graph graph;
graph.addConnection(connect(generateNumber, multiplyByTwo));
graph.addConnection(connect(multiplyByTwo, print));
Context ctx;
graph.execute(ctx); // выведет 84
2.6. Ограничения текущей реализации и пути их решения
Ограничение
Планируемое решение
Типы не проверяются (используется std::any)
Ввести систему типов (раздел 1.5) и автоматические конвертеры.
Нет топологической сортировки
Добавить построение графа зависимостей на основе метаданных компонентов (какие порты читаются/пишутся).
Копирование контекста при ветвлении может быть дорогим
Ввести семантику перемещения или использовать shared_ptr для больших данных.
Нет кэширования
Реализовать мемоизацию результатов узлов (раздел 1.6).
std::function имеет небольшие накладные расходы
Для критических участков можно использовать шаблонные компоненты и std::variant для хранения.


2.7. Вывод
Реализация протокола ArxGlue на C++ обеспечивает минимально необходимый функционал для декларативной композиции компонентов с поддержкой ветвления. Она служит фундаментом, на котором будут надстраиваться более сложные возможности: система типов, кэширование, многопоточность и интеграция с ассетами. Все эти расширения будут добавляться постепенно, сохраняя обратную совместимость с базовым протоколом.
Следующий раздел (3. Интеграция с ассетами) опишет, как встроить в эту модель загрузку внешних ресурсов и «запечку» процедурных данных.

3. Интеграция с ассетами
В классическом процедурном подходе (как в Werkkzeug4) весь контент генерируется «на лету» из графа. Однако для реальных игр и приложений часто требуется импортировать готовые данные: модели, текстуры, анимации, звуки. 3Tone поддерживает гибридный пайплайн, где ассеты и процедурные генераторы сосуществуют в одном графе, а сложные подграфы могут быть «запечены» (baked) в ассеты для оптимизации рантайма.
3.1. Узлы-загрузчики (LoadTexture, LoadMesh)
Узел-загрузчик — это компонент, который читает файл с диска (или из пакета ресурсов) и создаёт соответствующий объект (текстуру, меш, звук). Результат сохраняется в контексте и может быть использован другими узлами.
Пример реализации узла LoadTexture:
cpp
void LoadTexture(Context& ctx) {
    // Путь к файлу берётся из состояния (например, из параметров узла)
    std::string path = std::any_cast<std::string>(ctx.getState<std::string>("texturePath"));
    // Загрузка через ассет-менеджер (пока заглушка)
    auto texture = AssetManager::loadTexture(path);
    ctx.output = texture;
}
Для удобства создания таких узлов предусмотрена фабрика:
cpp
Component makeLoadTexture(const std::string& path) {
    return [path](Context& ctx) {
        ctx.output = AssetManager::loadTexture(path);
    };
}
Аналогично для мешей, звуков и других типов.
Важно: Загрузка ассета происходит при каждом выполнении узла, если результат не кэшируется. Поэтому обычно узел-загрузчик оборачивается в кэширующий узел или сам сохраняет загруженный объект в state и при повторных вызовах возвращает его без повторной загрузки.
3.2. Узлы-контейнеры (AssetNode)
Узел-контейнер (или AssetNode) — это компонент, который хранит ассет внутри себя (как часть своего состояния) и просто выдаёт его при вызове. В отличие от узла-загрузчика, он не обращается к диску, а использует уже загруженные данные, возможно, переданные через параметры или запечённые.
cpp
class AssetNode {
    std::any asset;
public:
    AssetNode(std::any a) : asset(a) {}
    void operator()(Context& ctx) {
        ctx.output = asset;
    }
};
Такие узлы могут быть созданы:
Вручную (например, при импорте ассета через редактор).
Автоматически в результате операции «запечки» (см. раздел 3.4).
Узел-контейнер не изменяет данные, он просто делает ассет доступным для других узлов графа.
3.3. Смешивание генераторов и ассетов
Главная идея: в одном графе могут быть как генераторы (Perlin noise, Mandelbrot, процедурные меши), так и загруженные ассеты. Более того, выход генератора может подаваться на вход узла-ассета (например, текстура шума накладывается на загруженную модель).
Пример: Создание текстуры, комбинирующей загруженное изображение с процедурным узором.
cpp
// Узел-загрузчик текстуры
auto loadBase = makeLoadTexture("base.png");
// Генератор шума (возвращает текстуру)
auto noise = makePerlinNoise(512, 512);
// Узел смешивания (blend)
auto blend = makeBlend(0.5f); // смешивает два входа с весами
// Соединения
Graph graph;
graph.addConnection(connect(loadBase, blend));     // loadBase -> blend (input1)
graph.addConnection(connect(noise, blend));        // noise -> blend (input2)
Context ctx;
graph.execute(ctx);
// Результат в ctx.output – скомбинированная текстура
Для реализации узла с несколькими входами (например, blend) используется доступ к state, где хранятся промежуточные результаты, или вводится расширенная система портов. В простейшем случае blend может читать два значения из state по ключам, а перед вызовом blend другие узлы записывают свои выходы в эти ключи.
3.4. «Запечка» (Bake) подграфа в ассет
Запечка — это процесс выделения подграфа, его выполнения и сохранения результата в виде ассета (текстуры, меша, звука). В исходном графе подграф заменяется узлом-контейнером, который ссылается на запечённый ассет. Это позволяет ускорить рантайм за счёт предварительного вычисления и уменьшить зависимость от сложных процедурных узлов.
Алгоритм запечки:
Пользователь выбирает выходной узел подграфа (корень).
Система определяет все узлы, от которых зависит этот выход (обход графа вглубь).
Создаётся временный граф, содержащий только выбранные узлы.
Выделяется контекст с начальными параметрами (если требуется).
Временный граф выполняется, результат сохраняется.
Результат экспортируется в файл ассета (например, baked_texture.png) или остаётся в памяти.
В исходном графе выделенный подграф заменяется узлом-контейнером, который загружает (или содержит) запечённый ассет.
Реализация (эскиз):
cpp
void bakeSubgraph(Graph& originalGraph, Component rootNode, const std::string& assetPath) {
    // 1. Построить список зависимостей rootNode (анализ соединений)
    std::vector<Component> subgraphNodes = collectDependencies(rootNode);
    
    // 2. Создать временный граф и выполнить его
    Graph tempGraph;
    for (auto& conn : originalGraph.getConnections()) {
        if (isInSubgraph(conn, subgraphNodes)) {
            tempGraph.addConnection(conn);
        }
    }
    Context tempCtx;
    tempGraph.execute(tempCtx);
    
    // 3. Сохранить результат в ассет (зависит от типа)
    saveAsset(tempCtx.output, assetPath);
    
    // 4. Заменить подграф в исходном графе на узел-контейнер
    Component assetNode = makeAssetNode(assetPath); // узел, загружающий ассет
    originalGraph.replaceSubgraph(rootNode, assetNode);
}
Параметры запечки:
Качество – для текстур можно указать сжатие (DXT), разрешение.
Формат – в какой тип ассета преобразовывать (PNG, JPG, собственный бинарный).
Статичность – если входы подграфа никогда не меняются, запечка даёт максимальный выигрыш.
Применение:
Запечка сложной процедурной текстуры в dds для быстрой загрузки в игре.
Преобразование генератора ландшафта в статический меш.
Фиксация анимации, сгенерированной из шума, в сэмплы.
3.5. Управление ресурсами и памятью
При смешивании генераторов и ассетов важно контролировать время жизни загруженных данных. Рекомендуется:
Использовать центральный менеджер ресурсов (AssetManager), который хранит загруженные ассеты в кэше и удаляет их при неиспользовании (счётчик ссылок).
Узлы-загрузчики и узлы-контейнеры должны увеличивать счётчик ссылок на ассет, когда выдают его в output, и уменьшать, когда перезаписывают выход (или при разрушении узла).
При запечке подграфа оригинальные узлы могут быть удалены, а ассет загружен через менеджер.
3.6. Пример: смешивание загруженной модели с процедурной анимацией
cpp
// Загружаем меш
auto loadMesh = makeLoadMesh("dragon.obj");
// Процедурная анимация (возвращает матрицу трансформации)
auto animator = makeWaveAnimation(0.5f);
// Узел, применяющий трансформацию к мешу (возвращает трансформированный меш)
auto transformMesh = makeTransformMesh();
// Соединения: loadMesh -> transformMesh (input1), animator -> transformMesh (input2)
Graph graph;
graph.addConnection(connect(loadMesh, transformMesh));
graph.addConnection(connect(animator, transformMesh));
Context ctx;
graph.execute(ctx);
// Результат – анимированный меш
Если анимация детерминирована и не зависит от времени, можно запечь несколько ключевых кадров в анимированный меш (формат MD5 или собственный).
3.7. Заключение
Интеграция ассетов в граф вычислений открывает возможности для гибридного пайплайна, где процедурные и импортированные данные работают вместе. Механизм запечки позволяет оптимизировать сложные подграфы, превращая их в эффективные ассеты для финального билда. В следующих разделах будет описана более сложная система типов и многопоточный исполнитель, которые усилят эту интеграцию.

4. Граф вычислений (вдохновлённый Werkkzeug4)
Граф вычислений — это сердце движка 3Tone. Он определяет, как компоненты (узлы) соединяются друг с другом, как вычисляются зависимости, как происходит многопоточное выполнение и как граф сохраняется на диск. В этом разделе мы опишем расширенную модель графа, вдохновлённую Werkkzeug4, но адаптированную под протокол ArxGlue.
4.1. Типы узлов
В рамках графа вычислений узлы классифицируются по их роли и поведению:
Генераторы (Source Nodes) – не имеют входных портов (или имеют только контекстные параметры). Примеры: LoadTexture, PerlinNoise, Constant.
Фильтры (Filter Nodes) – имеют один или несколько входов и один выход. Выполняют преобразование. Примеры: Blur, ColorBalance, TransformMesh.
Композиты (Composite Nodes) – объединяют несколько входов в один выход (например, Mix, Merge, Blend).
Разветвители (Split Nodes) – один вход, несколько выходов (хотя в протоколе ветвление реализуется через MultiConnection, для некоторых случаев полезны явные узлы-разветвители).
Контроллеры потока (Flow Control) – условное выполнение, циклы (расширение для будущих версий).
Контейнеры (Asset Nodes) – хранят и выдают ассеты (см. раздел 3.2).
Для каждого узла в графе могут быть определены метаданные:
Имя узла (для отладки и сериализации).
Тип узла (для GUI и оптимизаций).
Флаг Volatile – если установлен, узел никогда не кэшируется (например, генератор случайных чисел, чтение времени).
Флаг Pure – если установлен, выход узла зависит только от входов (не от глобального состояния). Это позволяет агрессивно кэшировать и переупорядочивать выполнение.
4.2. Топологическая сортировка и обновление зависимостей
В отличие от линейного графа из раздела 2.4, полноценная система должна поддерживать произвольные направленные ациклические графы (DAG). Входы узла могут быть соединены с выходами любых других узлов, независимо от порядка объявления.
Построение графа зависимостей:
Каждый узел имеет список входных портов (идентифицируемых по имени или индексу).
Каждое соединение связывает выходной порт одного узла с входным портом другого.
На основе соединений строится ориентированный граф: ребро A → B означает, что узел B зависит от узла A.
Для корректного выполнения необходимо выполнять узлы в топологическом порядке (предок раньше потомка).
Алгоритм топологической сортировки (Кан):
Вычислить для каждого узла количество входящих рёбер (степень захода).
Поместить в очередь все узлы с нулевой степенью захода.
Пока очередь не пуста:
Извлечь узел U.
Выполнить U (или просто добавить в список порядка).
Для каждого узла V, зависящего от U, уменьшить степень захода V. Если стала нулевой – добавить V в очередь.
Если после обработки остались узлы с ненулевой степенью – граф содержит цикл (ошибка).
Инкрементальное обновление:
При изменении параметров узла или входных данных не обязательно пересчитывать весь граф. Достаточно перевычислить затронутые узлы – те, которые находятся в подграфе, достижимом от изменённого узла.
Каждый узел хранит флаг dirty, который устанавливается при изменении любого из его входов.
При запросе выхода узла, если флаг dirty установлен, узел перевычисляется, затем флаг сбрасывается.
Для этого требуется, чтобы граф был ациклическим, иначе возможны бесконечные циклы.
4.3. Многопоточный исполнитель (планировщик задач)
Для повышения производительности на многоядерных системах граф должен выполняться параллельно. Однако не все узлы можно выполнять одновременно – необходимо соблюдать зависимости.
Планировщик на основе пула потоков (адаптированный из Werkkzeug4):
На этапе подготовки строится топологический порядок, но в отличие от последовательного выполнения, узлы, не имеющие зависимостей друг от друга, могут выполняться параллельно.
Используется счётчик зависимостей: каждый узел хранит количество неподготовленных предков. Когда предок завершает выполнение, счётчик потомка уменьшается. Когда счётчик становится нулевым, потомок отправляется в пул потоков.
Для эффективного использования кэша данные (входы/выходы) должны быть потокобезопасными. Проще всего использовать std::shared_ptr и копирование данных, но для больших ассетов (текстуры, меши) следует применять shared_ptr с подсчётом ссылок и неизменяемость (copy-on-write).
Планируемая архитектура исполнителя:
cpp
class Executor {
    ThreadPool pool;
    Graph* graph;
    std::vector<Node*> topoOrder;       // топологический порядок (может быть несколько)
    std::atomic<int> remainingTasks;
public:
    void execute(Graph& g, Context& ctx);
};
Алгоритм:
Построить топологический порядок узлов (один раз, при изменении структуры графа).
Для каждого узла определить количество предков (inDegree).
Инициализировать очередь готовых узлов (тех, у которых inDegree == 0).
Для каждого узла из очереди отправить задачу в пул потоков:
Выполнить узел (вызвать его компонент с контекстом).
Для каждого потомка атомарно уменьшить inDegree. Если стал 0 – добавить в очередь.
Дождаться завершения всех задач.
Оптимизации:
Группировка узлов – если несколько узлов подряд не имеют ветвлений и их выполнение дешёво, можно объединить в одну задачу, чтобы снизить оверхед.
Кэширование результатов – если узел уже был вычислен ранее и его входы не изменились, пропускаем выполнение (даже в многопоточном режиме).
Affinity – для узлов, работающих с GPU (рендеринг), может быть полезно привязать их к основному потоку или выделенному потоку.
4.4. Сериализация графа (JSON / бинарный протокол)
Для сохранения и загрузки графа (в редакторе, для обмена, для финального билда) необходима сериализация. Поддерживаются два формата: JSON (читаемый, для разработки) и бинарный (быстрый, для рантайма).
Что сериализуется:
Список узлов, для каждого:
Уникальный идентификатор (или имя типа).
Параметры (значения, пути к файлам, числа, строки).
Позиция в редакторе (опционально).
Список соединений (от какого узла и порта к какому узлу и порту).
Глобальные настройки графа.
Пример JSON-представления:
json
{
  "version": 1,
  "nodes": [
    {
      "id": 1,
      "type": "LoadTexture",
      "params": { "path": "textures/grass.png" }
    },
    {
      "id": 2,
      "type": "PerlinNoise",
      "params": { "scale": 0.01, "octaves": 4 }
    },
    {
      "id": 3,
      "type": "Blend",
      "params": { "factor": 0.7 }
    }
  ],
  "connections": [
    { "src": 1, "srcPort": 0, "dst": 3, "dstPort": 0 },
    { "src": 2, "srcPort": 0, "dst": 3, "dstPort": 1 }
  ]
}
Реализация сериализации в C++:
Каждый тип узла регистрирует свою фабрику и функцию сохранения/загрузки параметров.
Используется библиотека nlohmann/json или собственный бинарный поток.
При загрузке графа создаются узлы через фабрики, затем восстанавливаются соединения.
4.5. Пример: построение графа с зависимостями и многопоточное выполнение
cpp
// Создаём узлы
auto constant1 = makeConstant(10);
auto constant2 = makeConstant(20);
auto add = makeAdd();       // складывает два входа
auto multiply = makeMultiply(); // умножает два входа

// Строим граф: add = const1 + const2, multiply = add * const1
Graph g;
g.addNode(constant1);
g.addNode(constant2);
g.addNode(add);
g.addNode(multiply);

g.connect(constant1, add, 0, 0);   // выход constant1 -> вход add[0]
g.connect(constant2, add, 0, 1);   // выход constant2 -> вход add[1]
g.connect(add, multiply, 0, 0);    // выход add -> вход multiply[0]
g.connect(constant1, multiply, 0, 1); // выход constant1 -> вход multiply[1]

// Выполнение с многопоточным исполнителем
Executor exec;
Context ctx;
exec.execute(g, ctx);
int result = any_cast<int>(ctx.getState("multiply_output")); // 30*10 = 300
В данном примере add и multiply не зависят друг от друга по данным (кроме последовательности), но из-за зависимости multiply от add они выполнятся последовательно. Однако если бы были два независимых подграфа, они бы выполнялись параллельно.
4.6. Расширения: условное выполнение и циклы
Хотя базовый граф предполагает статическую структуру, для некоторых задач (например, обработка списка объектов, итеративная генерация) необходимы циклы и условные переходы. В будущем можно добавить:
Узел-условие (Condition): имеет вход-условие (bool) и два выхода (true / false). Поток выполнения направляется по одному из путей.
Узел-цикл (Loop): выполняет подграф до тех пор, пока условие истинно. Реализуется через обратную связь (что нарушает DAG). Для поддержки циклов потребуется специализированный исполнитель, который может выполнять подграф итеративно.
Однако для большинства procedural content generation задач DAG достаточно. Циклы можно эмулировать через рекурсивные вызовы подграфов с ограниченным числом итераций.
4.7. Заключение
Раздел описал расширенную модель графа вычислений, вдохновлённую Werkkzeug4: типы узлов, топологическую сортировку, многопоточный исполнитель и сериализацию. Эти механизмы обеспечивают масштабируемость, производительность и гибкость, необходимые для создания сложных процедурных пайплайнов. В следующем разделе (5. Рендеринг и представление данных) мы рассмотрим, как результаты вычислений визуализируются на экране.

5. Рендеринг и представление данных
В движке 3Tone рендеринг рассматривается как особый тип вычислений — преобразование данных (текстур, мешей, материалов) в изображение на экране или в текстуру. Для обеспечения кроссплатформенности и производительности рендеринг вынесен в отдельный абстрактный слой, а в графе вычислений присутствуют специальные узлы вывода, которые инициируют отрисовку.
5.1. Абстрактный графический слой (Vulkan / DirectX 12 / Metal)
Вместо прямого использования конкретного API (DirectX 9/11, как в Werkkzeug4), 3Tone строит свой рендеринг на основе абстрактного графического бэкенда. Это позволяет в будущем поддерживать несколько API, а также упрощает тестирование и портирование.
Ключевые абстракции:
RenderDevice – представление устройства (адаптер, контекст, очередь команд).
Texture – текстура (2D, Cube, 3D) с методами загрузки, копирования, привязки.
Buffer – вершинный/индексный/константный буфер.
Shader – программа (вершинный, пиксельный, compute шейдер).
Pipeline – состояние рендеринга (растеризатор, blending, depth-stencil).
CommandBuffer – набор команд (рисование, смена состояний).
Выбор бэкенда:
Для Windows: DirectX 12 (максимальная производительность) или Vulkan.
Для Linux: Vulkan.
Для macOS/iOS: Metal.
Для прототипирования: программный растеризатор или OpenGL (устаревший, но простой).
Реализация бэкенда скрыта за интерфейсами, а конкретные узлы графа работают только с абстракциями. Это позволяет менять графический API без переписывания узлов.
5.2. Узлы вывода (RenderScene, RenderToTexture)
Для интеграции рендеринга в граф предусмотрены специальные узлы, которые принимают на вход сцену (или отдельные меши/материалы) и выдают изображение (текстуру). Эти узлы являются листьями графа – они не имеют выходов, но могут модифицировать глобальный контекст (например, устанавливать текущий кадровый буфер).
Узел RenderScene:
Входы: сцена (Scene), камера (Camera), параметры окружения (Env).
Выход: отсутствует (но можно добавить текстуру результата).
Действие: выполняет рендеринг сцены в текущий целевой буфер (обычно экран или текстуру, заданную в контексте).
cpp
void renderSceneNode(Context& ctx) {
    Scene* scene = any_cast<Scene*>(ctx.input);
    Camera* camera = any_cast<Camera*>(ctx.getState("camera"));
    RenderTarget* target = any_cast<RenderTarget*>(ctx.getState("renderTarget"));
    
    // Получить бэкенд из глобального менеджера
    auto device = GraphicsBackend::getDevice();
    CommandBuffer cmd = device->createCommandBuffer();
    cmd.begin();
    // Установить цель, очистить, нарисовать сцену
    cmd.setRenderTarget(target);
    cmd.clear({0.2f, 0.2f, 0.2f, 1.0f});
    scene->draw(cmd, camera);
    cmd.end();
    device->submit(cmd);
}
Узел RenderToTexture:
Входы: сцена, камера.
Выход: текстура (результат рендеринга).
Применение: рендеринг в текстуру для дальнейшего использования (например, пост-эффекты, reflection maps).
cpp
void renderToTextureNode(Context& ctx) {
    Scene* scene = any_cast<Scene*>(ctx.input);
    Camera* camera = any_cast<Camera*>(ctx.getState("camera"));
    auto device = GraphicsBackend::getDevice();
    
    // Создать или получить текстуру из кэша
    Texture* target = device->createTexture(width, height, Format::RGBA8, Usage::RenderTarget);
    CommandBuffer cmd = device->createCommandBuffer();
    cmd.begin();
    cmd.setRenderTarget(target);
    cmd.clear(...);
    scene->draw(cmd, camera);
    cmd.end();
    device->submit(cmd);
    ctx.output = target;
}
Узел PostProcess – применяет эффект к текстуре (размытие, тональная коррекция, bloom). Вход: текстура. Выход: обработанная текстура.
5.3. Визуализация графа в редакторе (интеграция с ImGui или Qt)
Для удобства разработки графа необходим визуальный редактор, где узлы представлены блоками, а соединения — линиями. 3Tone предоставляет два варианта:
Встроенный редактор на базе Dear ImGui – легковесный, легко интегрируется в движок, но имеет ограниченные возможности кастомизации.
Отдельное приложение на Qt – более мощное, с поддержкой вкладок, док-панелей, экспорта/импорта, но требует отдельного процесса.
Основные функции редактора:
Добавление узлов из палитры (по типам).
Соединение портов (перетаскиванием мыши).
Редактирование параметров узла (инспектор свойств).
Отображение предпросмотра (для текстур, мешей, результатов).
Сохранение/загрузка графа (сериализация в JSON/бинарный).
Отладка: подсветка «грязных» узлов, пошаговое выполнение.
Интеграция с ImGui (пример):
cpp
class NodeEditor {
    ImNodes::EditorContext* context;
    Graph* graph;
public:
    void render() {
        ImNodes::BeginNodeEditor();
        for (auto& node : graph->getNodes()) {
            ImNodes::BeginNode(node.id);
            // Заголовок, порты, содержимое
            for (auto& input : node.inputs) {
                ImNodes::BeginInputAttribute(input.id);
                ImGui::Text(input.name.c_str());
                ImNodes::EndInputAttribute();
            }
            // ... параметры
            for (auto& output : node.outputs) {
                ImNodes::BeginOutputAttribute(output.id);
                ImGui::Text(output.name.c_str());
                ImNodes::EndOutputAttribute();
            }
            ImNodes::EndNode();
        }
        ImNodes::EndNodeEditor();
        // Обработка новых соединений
        if (ImNodes::IsLinkCreated(...)) {
            graph->connect(...);
        }
    }
};
5.4. Представление данных для рендеринга
Чтобы узлы могли обмениваться графическими данными (текстурами, мешами), необходимо унифицированное представление, независимое от бэкенда.
Texture – абстрактный класс с методами getWidth(), getHeight(), getFormat(), bind(slot). Внутри может хранить дескриптор Vulkan, ID3D12Resource или GLuint.
Mesh – содержит вершинные буферы, индексный буфер, bounding box. Методы draw(CommandBuffer&).
Material – ссылки на шейдеры и текстуры, параметры (цвет, коэффициент).
Scene – иерархия узлов (ноды с трансформациями, меши, материалы, камеры, источники света).
Эти классы реализованы поверх абстрактного графического слоя и не зависят от конкретного API.
5.5. Пример: рендеринг сцены с пост-эффектом
cpp
// Узел-загрузчик меша
auto loadMesh = makeLoadMesh("dragon.obj");
// Узел-загрузчик текстуры
auto loadTexture = makeLoadTexture("dragon_diffuse.png");
// Узел-материал (принимает меш и текстуру, выдаёт готовый к рендерингу объект)
auto material = makeMaterial(loadMesh, loadTexture);
// Узел-камера (константа)
auto camera = makeConstantCamera(Vector3(0,0,5), Vector3(0,0,0));
// Узел-сцена (группирует)
auto scene = makeScene(material, camera);
// Узел рендеринга в текстуру
auto renderToTex = makeRenderToTexture(scene, camera);
// Узел пост-эффекта (размытие)
auto blur = makeBlur(renderToTex, 5.0f);
// Узел вывода на экран
auto output = makeRenderToScreen(blur);

Graph graph;
graph.addConnection(connect(loadMesh, material, 0));
graph.addConnection(connect(loadTexture, material, 1));
graph.addConnection(connect(material, scene, 0));
graph.addConnection(connect(camera, scene, 1));
graph.addConnection(connect(scene, renderToTex));
graph.addConnection(connect(renderToTex, blur));
graph.addConnection(connect(blur, output));

Context ctx;
graph.execute(ctx); // отрисовывает сцену с размытием на экран
5.6. Оптимизации рендеринга в графе
Frustum culling – если камера известна, узлы-меши могут проверять видимость и не отправлять невидимые объекты в рендер.
Batch-рендеринг – несколько мешей с одинаковым материалом можно объединить в один draw call. Узел BatchScene выполняет эту оптимизацию.
Рендеринг в несколько проходов – для теней, отражений, пост-эффектов граф может содержать несколько узлов RenderToTexture, соединённых последовательно.
Асинхронный рендеринг – рендеринг может выполняться в отдельном потоке, а граф вычислений – в другом. Для синхронизации используются семафоры или fence.

В следующем разделе (6. Инструменты разработчика) будут рассмотрены отладчик, профилировщик и средства экспорта ассетов.

6. Инструменты разработчика
Для эффективной работы с графом вычислений и отладки сложных процедурных пайплайнов 3Tone предоставляет набор инструментов: визуальный редактор графа, отладчик с пошаговым выполнением, профилировщик производительности узлов и средства экспорта ассетов. Все инструменты интегрированы в среду разработки (отдельное приложение или режим редактора внутри движка).
6.1. Редактор графа (Node Editor)
Редактор графа — основное рабочее пространство разработчика. Он позволяет создавать, удалять, соединять узлы, редактировать их параметры и наблюдать за результатами.
Функциональность:
Палитра узлов – список доступных типов узлов, сгруппированных по категориям (генераторы, фильтры, ассеты, рендеринг). Поиск по имени.
Рабочая область – бесконечное полотно с возможностью масштабирования и панорамирования. Узлы отображаются в виде блоков с портами (входы слева, выходы справа).
Создание соединений – перетаскиванием от порта-источника к порту-приёмнику. Система проверяет совместимость типов и автоматически предлагает конвертеры, если необходимо.
Инспектор свойств – панель с параметрами выбранного узла (числа, строки, флаги, ссылки на файлы). Изменения применяются к графу в реальном времени.
Предпросмотр – для узлов, выдающих визуальные данные (текстуры, меши), можно открыть окно предпросмотра, обновляющееся при каждом вычислении.
Комментарии и аннотации – возможность добавлять текстовые блоки, группировать узлы в «регионы» для удобства навигации.
История изменений – поддержка undo/redo для всех операций над графом.
Реализация:
Редактор может быть реализован с использованием библиотеки ImNodes (для ImGui) или Qt Graphics View Framework. В 3Tone планируется использовать ImNodes как более лёгкую интеграцию с движком.
cpp
class GraphEditor {
    Graph* graph;
    ImNodesEditorContext* editorCtx;
    std::unordered_map<NodeId, NodeUI> nodeUI;
public:
    void render() {
        ImNodes::BeginNodeEditor();
        for (auto& node : graph->nodes) {
            renderNode(node);
        }
        ImNodes::EndNodeEditor();
        handleConnections();
    }
    void renderNode(Node& node) {
        ImNodes::BeginNode(node.id);
        ImNodes::BeginNodeTitleBar();
        ImGui::Text(node.name.c_str());
        ImNodes::EndNodeTitleBar();
        // Отрисовка портов и параметров
        // ...
        ImNodes::EndNode();
    }
};
6.2. Отладчик (пошаговое выполнение, просмотр значений)
Отладчик позволяет приостановить выполнение графа, просмотреть промежуточные значения и выполнять узлы по шагам. Это критически важно для поиска ошибок в сложных зависимостях.
Возможности:
Точки останова – установка на любом узле. При достижении узла выполнение графа останавливается.
Пошаговое выполнение – шаг внутрь (в составной узел), шаг обход (выполнить узел и перейти к следующему), продолжить до следующей точки.
Просмотр значений – при останове можно посмотреть input, output и state любого узла. Поддерживаются разные представления (числа, строки, текстуры – в виде миниатюры).
Трассировка потока данных – визуальное отображение, какие значения передавались по каждому соединению (появляется подсказка с типом и кратким содержимым).
Условные точки останова – останов, только если значение узла удовлетворяет условию (например, output > 100).
Реализация:
Для поддержки отладки исполнитель графа (Executor) должен быть модифицирован:
Возможность выполнять граф в режиме «под отладчиком» – перед вызовом узла проверять наличие точки останова и приостанавливать поток.
Сохранять историю выполненных узлов и их выходных значений (ограниченный размер).
Предоставлять API для интроспекции состояния.
cpp
class DebugExecutor : public Executor {
    std::set<NodeId> breakpoints;
    bool stepping;
    NodeId currentStepNode;
public:
    void setBreakpoint(NodeId id) { breakpoints.insert(id); }
    void stepInto() { stepping = true; }
    void execute(Graph& g, Context& ctx) override {
        // Модифицированный алгоритм с проверками
    }
};
6.3. Профилирование производительности узлов
Для оптимизации процедурных генераций важно знать, какие узлы самые медленные. Профилировщик собирает статистику выполнения и отображает её в удобной форме.
Собираемые метрики:
Время выполнения – общее и среднее (в микросекундах) для каждого узла.
Количество вызовов – сколько раз узел был выполнен за один проход графа.
Время ожидания – если узел многопоточный, время, проведённое в очереди.
Использование памяти – сколько данных кэширует узел.
Отображение:
В редакторе графа узлы могут быть подсвечены цветом в зависимости от времени выполнения (красный – медленный, зелёный – быстрый).
Детальная статистика доступна в отдельной панели в виде таблицы или гистограммы.
Возможность экспорта профиля в JSON для дальнейшего анализа.
Интеграция:
В исполнителе замеряется время через std::chrono::high_resolution_clock или платформенно-зависимые таймеры (QueryPerformanceCounter на Windows). Сбор статистики можно включать/выключать через флаг.
cpp
struct ProfilerData {
    std::string nodeName;
    uint64_t totalTimeNs;
    uint64_t callCount;
    uint64_t maxTimeNs;
    void addSample(uint64_t ns) { totalTimeNs += ns; callCount++; maxTimeNs = std::max(maxTimeNs, ns); }
};
std::unordered_map<NodeId, ProfilerData> profileData;
6.4. Экспорт ассетов и «запечка» для релиза
Финальный этап разработки – экспорт проекта в ассеты, готовые для загрузки в рантайм игры. Здесь на помощь приходит механизм запечки (baking), описанный в разделе 3.4.
Инструменты экспорта:
Bake Subgraph – выбрать выходной узел, указать формат ассета (PNG, DDS, OBJ, FBX, собственный бинарный) и путь. Система вычисляет подграф и сохраняет результат.
Batch Bake – для множества узлов (например, все текстуры в проекте) запускается фоновая задача, которая последовательно запекает их, позволяя разработчику продолжать работу.
Optimization – при запечке можно применить сжатие (DXT для текстур, quantisation для мешей), удалить неиспользуемые каналы.
Экспорт всего графа как «чёрного ящика»:
В некоторых случаях удобно сохранить весь граф в бинарном виде для загрузки в рантайме (например, для обновляемого контента). Тогда рантайм содержит упрощённый исполнитель без редактора, но способный выполнять загруженный граф.
Интеграция с системой сборки:
3Tone может генерировать файлы ассетов в рамках CI/CD пайплайна. Для этого предусмотрен консольный режим движка: 3tone --bake project.json --output assets/.
bash
# Пример командной строки для запечки
3tone-cli bake --graph scene_graph.json --node "FinalRender" --format png --output final.png
6.5. Дополнительные инструменты
Валидатор графа – проверка на циклы, висячие соединения, несовместимые типы. Выдаёт предупреждения и ошибки.
Генератор кода – для статических подграфов можно сгенерировать C++ код, который выполняет ту же логику без интерпретации графа (для максимальной производительности).
Сравнение версий – для графов, хранящихся в системе контроля версий, полезен diff-инструмент, показывающий изменения в структуре и параметрах.
6.6. Заключение
Инструменты разработчика делают 3Tone не просто движком, а полноценной средой для создания процедурного контента. Редактор графа, отладчик и профилировщик позволяют быстро итеративно разрабатывать сложные генерации, а механизмы экспорта и запечки обеспечивают эффективный пайплайн для финальных продуктов.
7. Стандарты и протоколы
При разработке 3Tone несколькими разработчиками критически важно соблюдать единые стандарты и протоколы взаимодействия. Этот раздел фиксирует обязательные для всех участников проекта правила, интерфейсы и форматы, обеспечивающие совместимость разрабатываемых модулей.
7.1. Общие стандарты кода
7.1.1. Язык и компилятор
C++17 (минимальная версия). Допускается использование C++20 в изолированных модулях при согласовании.
Компиляторы: MSVC (Windows), Clang (Linux/macOS), AppleClang (macOS).
Единый файл конфигурации .clang-format для автоматического форматирования (см. приложение).

7.1.3. Организация кода
Каждый логический модуль (ядро графа, протокол, рендеринг) — в отдельной директории src/<module>/.
Публичные заголовки — в include/3tone/<module>/.
Пространства имён:
arxglue – протокол компонентов и контекст.
core – граф, узлы, исполнитель.
render – графический бэкенд, рендеринг.
editor – инструменты редактора.
assets – работа с ассетами и запечка.
7.1.4. Обработка ошибок
Исключения – для критических ошибок (неверный формат файла, отсутствие графического устройства). Базовый класс исключений: 3tone::Exception.
Коды возврата – для ожидаемых ошибок (невалидный узел, несовместимые типы). Использовать std::error_code или пользовательские перечисления.
Ассерты – только в отладочных сборках. Не использовать для проверки пользовательского ввода.
7.2. Протокол компонента (ArxGlue)
Каждый компонент должен удовлетворять следующему контракту:
Сигнатура: void(Context&). Компонент не может бросать исключения, не предусмотренные контрактом (допускаются только std::bad_any_cast при ошибках ввода).
Чтение входа: Компонент может читать ctx.input (тип std::any). Если ожидается конкретный тип, должен выполнять any_cast и корректно обрабатывать исключение (устанавливать флаг ошибки в контексте).
Запись выхода: Компонент обязан записать результат в ctx.output. Если результат не может быть вычислен, должен записать пустое значение (std::any{}) и установить флаг ошибки.
Использование состояния: Компонент может читать и писать в ctx.state. Ключи должны быть уникальны для компонента или документированы. Рекомендуется использовать префикс, совпадающий с именем компонента (например, "PerlinNoise::seed").
Чистота (pure): Если компонент детерминированно зависит только от input и state (но не от глобального времени, случайных чисел), он может быть помечен флагом pure = true для оптимизаций.
Пример декларации компонента с метаинформацией:
cpp
struct ComponentMetadata {
    std::string name;
    std::vector<PortInfo> inputs;   // имена и типы ожидаемых портов (опционально)
    std::vector<PortInfo> outputs;
    bool pure;
    bool volatile_;
};
Каждый компонент должен предоставлять функцию getMetadata(), возвращающую ComponentMetadata. Это позволяет редактору и исполнителю принимать решения о кэшировании и проверке типов.
7.3. Протокол взаимодействия графа и исполнителя
7.3.1. Интерфейс INode
Базовый класс для всех узлов графа:
cpp
class INode {
public:
    virtual ~INode() = default;
    virtual void execute(Context& ctx) = 0;
    virtual ComponentMetadata getMetadata() const = 0;
    virtual std::vector<Connection> getConnections() const = 0;
    virtual void setParameter(const std::string& name, const std::any& value) = 0;
    virtual std::any getParameter(const std::string& name) const = 0;
    // Сериализация
    virtual void serialize(nlohmann::json& j) const = 0;
    virtual void deserialize(const nlohmann::json& j) = 0;
};
7.3.2. Контракт исполнителя (IExecutor)
cpp
class IExecutor {
public:
    virtual ~IExecutor() = default;
    // Выполнить граф, начиная с указанных выходных узлов (если пусто – все корневые)
    virtual void execute(Graph& graph, Context& ctx, const std::vector<NodeId>& rootNodes = {}) = 0;
    // Принудительно инвалидировать кэш узла
    virtual void invalidate(NodeId id) = 0;
    // Регистрация внешнего провайдера ресурсов (для загрузки ассетов)
    virtual void setResourceProvider(IResourceProvider* provider) = 0;
};
Важно: Исполнитель не должен изменять структуру графа (добавлять/удалять узлы). Только вычислять значения.
7.3.3. Потокобезопасность
Исполнитель может вызывать execute() из нескольких потоков одновременно, но для одного графа должен быть обеспечен внешний мьютекс.
Компоненты не обязаны быть потокобезопасными, если они не помечены как thread_safe = true. Исполнитель гарантирует, что один узел не будет выполняться параллельно сам с собой, но разные узлы могут выполняться параллельно.
7.4. Стандарт сериализации
Графы, узлы и ассеты должны сохраняться в JSON (для разработки) или бинарном формате CBOR (для рантайма). Обязательные поля для каждого узла:
json
{
  "version": 1,
  "nodes": [
    {
      "id": 123,
      "type": "LoadTexture",
      "params": { "path": "data/tex.png" },
      "position": { "x": 100, "y": 200 },
      "metadata": { "comment": "diffuse map" }
    }
  ],
  "connections": [
    { "src": 123, "srcPort": 0, "dst": 456, "dstPort": 1 }
  ]
}
Идентификаторы узлов (id) должны быть уникальны в пределах файла.
type – строка, по которой фабрика узлов может создать экземпляр.
params – произвольный объект, специфичный для типа узла. Типы значений: int, float, string, bool, array, object.
position – для редактора (не влияет на вычисления).
connections – список соединений между узлами.
Стандарт для бинарного формата (CBOR) – используется та же схема, но с кодированием по правилам CBOR (RFC 8949). Для обеспечения обратной совместимости все сериализаторы должны игнорировать неизвестные поля.
7.5. Система плагинов (динамическая загрузка узлов)
Чтобы разные разработчики могли добавлять свои типы узлов без изменения ядра, 3Tone поддерживает плагины в виде динамических библиотек (.dll / .so / .dylib).
Контракт плагина:
cpp
extern "C" {
    // Функция, возвращающая метаинформацию о плагине
    PluginInfo getPluginInfo();
    // Функция, регистрирующая фабрики узлов в реестре
    void registerNodes(NodeRegistry& registry);
    // Опционально: инициализация и деинициализация плагина
    void initPlugin(PluginContext& ctx);
    void shutdownPlugin();
}
NodeRegistry – синглтон, позволяющий зарегистрировать функцию создания узла по строковому идентификатору:
cpp
registry.registerFactory("PerlinNoise", []() -> std::unique_ptr<INode> {
    return std::make_unique<PerlinNoiseNode>();
});
Плагины размещаются в директории plugins/ и загружаются при старте редактора/движка.
7.6. Протокол передачи данных между потоками
В многопоточном исполнителе данные передаются через Context, который содержит std::any. Чтобы избежать гонок:
Правило: Каждый узел получает собственный экземпляр контекста (или копию). Модификации не видны другим узлам, выполняющимся параллельно.
Для глобальных данных (например, загруженные текстуры) используется shared_ptr к неизменяемому объекту. Модификация таких данных должна создавать новую копию (copy-on-write).
Очереди сообщений – если узлы должны обмениваться событиями (например, прогресс загрузки), используется concurrent_queue из библиотеки (например, moodycamel::ConcurrentQueue).
7.7. Стандарт логирования и диагностики
Все модули должны использовать единый макрос для логирования:
cpp
#define 3TONE_LOG(level, message) \
    Logger::instance().log(level, __FILE__, __LINE__, message)
Уровни: DEBUG, INFO, WARNING, ERROR, FATAL.
Логгер по умолчанию выводит в консоль, но может быть перенаправлен в файл или в окно редактора. Плагины не должны использовать собственные логгеры.

7.8. Заключение
Соблюдение описанных стандартов и протоколов обеспечит беспроблемную интеграцию кода, написанного разными разработчиками, и долгосрочную поддерживаемость проекта 3Tone. Все изменения в этих стандартах должны фиксироваться в документе и согласовываться командой.


8. Примеры использования
Этот раздел демонстрирует практические сценарии работы с движком 3Tone. Примеры охватывают создание процедурных текстур, ландшафтов, объектов и анимаций, а также показывают, как модернизировать существующие ассеты (префабы) с помощью генераторов и смешивать их с процедурными данными.
Все примеры используют синтаксис графа на C++ (через API) или псевдокод для ясности. В реальном редакторе операции выполняются визуально, но логика идентична.
8.1. Процедурная генерация текстур
8.1.1. Генерация текстуры шума Перлина
Создадим текстуру размером 1024×1024 с 4 октавами шума Перлина и сохраним её в ассет.
Граф:
text
[ConstantInt: 1024] ───┐
[ConstantInt: 1024] ───┼──> [PerlinNoise: scale=0.01, octaves=4] ───> [SaveTexture: "noise.png"]
[ConstantInt: 4]      ───┘
Реализация на C++:
cpp
auto width = makeConstantInt(1024);
auto height = makeConstantInt(1024);
auto octaves = makeConstantInt(4);
auto perlin = makePerlinNoise(width, height, octaves, 0.01f);
auto saver = makeSaveTexture(perlin, "noise.png");

Graph g;
g.addConnection(connect(width, perlin, 0));
g.addConnection(connect(height, perlin, 1));
g.addConnection(connect(octaves, perlin, 2));
g.addConnection(connect(perlin, saver));

Context ctx;
g.execute(ctx);
8.1.2. Динамическая текстура с пост-эффектами

Предположим, у нас есть префаб «старый дом» (загруженный ассет). Мы хотим добавить к нему процедурно генерируемые элементы: текстуру с облупившейся краской, сгенерированную крышу и анимацию дыма из трубы.
Исходный префаб: old_house.obj + brick_diffuse.png.
Граф модернизации:

8.6. Смешивание ассета с генератором в реальном времени
Пример: игра, где игрок может изменять параметры генератора, влияющие на загруженную модель. Скажем, оружие (ассет) динамически покрывается процедурной коррозией.
Граф:

8.7. Гибридный пайплайн: запечка + динамическое обновление
Сложный подграф (например, генератор города) запекается в ассет для релизной версии, но в редакторе остаётся живым для экспериментов.
Режим разработки (редактор):
text
[CityGenerator] ───> [RenderView]
Режим релиза (запечённый ассет):
text
[LoadAsset: "city.bin"] ───> [RenderView]
При этом запечка выполняется командой:
bash
3tone-cli bake --graph city_gen.json --node CityOutput --output city.bin

Раздел 9. Компоненты, Узлы, Вложенность
Движок 3tone предоставляет гибридную модель построения вычислительных графов, сочетающую объектно-ориентированные узлы (INode) и легковесные функциональные компоненты (Component). Такая архитектура позволяет создавать сложные процедурные системы с произвольным уровнем вложенности, не жертвуя производительностью и гибкостью.
9.1. Два уровня абстракции: INode и Component
9.1.1. INode — объектно-ориентированный узел
INode — это базовый класс для всех узлов, выполняемых планировщиком Executor в основном графе Graph. Узлы обладают:
Метаданными (ComponentMetadata): имя, описание входных и выходных портов, флаги чистоты (pure) и волатильности (volatile_).
Кэшированием: результаты выполнения могут сохраняться и использоваться повторно при неизменных входах и параметрах.
Сериализацией: узлы можно сохранять в JSON и восстанавливать.
Потокобезопасностью: при правильной настройке метаданных Executor может выполнять узлы параллельно.
INode подходит для построения каркаса приложения: крупных вычислительных блоков, работы с ассетами, рендеринга.
9.1.2. Component — функциональный блок
Component — это просто функция с сигнатурой void(Context&), объявленная в arxglue.hpp. Она не имеет встроенного кэширования или метаданных, но может свободно читать и писать данные в общий Context. Компоненты легко комбинируются в цепочки с помощью структур MultiConnection и функции connect (см. arxglue.hpp).
Главное преимущество компонентов — быстрота создания и минимальные накладные расходы. Их удобно использовать внутри узла для реализации внутренней логики без создания десятков мелких INode.
9.2. Контекст выполнения и вложенность
9.2.1. Глобальный и локальный контексты
Глобальный контекст передаётся планировщиком Executor всем узлам основного графа. Он содержит:
input / output — основные входное и выходное значения.
state — потокобезопасное хранилище std::any с доступом по строковому ключу.
Локальный контекст может быть создан внутри любого узла для выполнения изолированного подграфа. Данные в локальном контексте не видны снаружи, что предотвращает конфликты имён и обеспечивает безопасную рекурсию.
9.2.2. Уровни вложенности
Движок не накладывает ограничений на глубину вложенности графов. Можно чередовать INode-графы и Component-цепочки произвольным образом:
Узел верхнего уровня (0) содержит компонентную цепочку (уровень 1).
Один из компонентов оборачивает и выполняет другой INode-граф (уровень 2).
Этот граф внутри снова использует компоненты (уровень 3) и так далее.
На практике глубина ограничена только объёмом памяти и здравым смыслом разработчика.
9.3. Использование локальных графов внутри INode
Часто возникает необходимость внутри одного узла выполнить небольшой подграф, не добавляя его в основной граф. Для этого достаточно создать временный экземпляр Graph, заполнить его узлами и выполнить с отдельным Context.
Пример: узел, генерирующий одну октаву шума с размытием.
cpp
class OctaveNode : public INode {
public:
    void execute(Context& ctx) override {
        // ... получение входных параметров ...
        Graph localGraph;
        auto noiseNode = std::make_unique<NoiseNode>();
        NodeId noiseId = localGraph.addNode(std::move(noiseNode));
        auto blurNode = std::make_unique<BlurNode>();
        NodeId blurId = localGraph.addNode(std::move(blurNode));
        localGraph.addConnection({noiseId, 0, blurId, 0});

        Context localCtx;
        // Передача входных значений через localCtx.state
        Executor exec(1);
        exec.execute(localGraph, localCtx, {blurId});

        // Результат из localCtx.output передаём на выход узла
        setOutputValue(ctx, 0, localCtx.output);
    }
};
Такой подход полностью изолирует подграф от основного контекста и позволяет легко повторно использовать существующие узлы внутри других узлов.
9.4. Функциональные цепочки компонентов
Когда не требуется кэширование промежуточных результатов или метаданные, компоненты позволяют быстро собрать конвейер обработки данных.
Пример: размытие текстуры как компонент.
cpp
auto blurComponent = [](Context& ctx) {
    auto tex = std::any_cast<std::shared_ptr<TextureAsset>>(ctx.state["input_tex"]);
    int radius = std::any_cast<int>(ctx.state["blur_radius"]);
    auto result = blurTexture(tex, radius);
    ctx.state["blurred_tex"] = result;
};
Компоненты могут быть скомбинированы с помощью connect:
cpp
auto step1 = [](Context& c) { /* подготовка */ };
auto step2 = [](Context& c) { /* обработка */ };
auto chain = connect(step1, step2);
9.5. Обёртка узла в компонент (NodeComponent)
Иногда нужно выполнить отдельный INode внутри компонентной цепочки. Для этого служит шаблонный класс NodeComponent (может быть добавлен в утилиты движка). Он создаёт временный граф, подключает входные и выходной узлы и возвращает результат как значение.
Пример использования:
cpp
auto constNode = std::make_unique<ConstantNode<int>>(100);
NodeComponent<int> constComp(std::move(constNode));
int val = constComp(ctx); // val == 100
Для узлов с несколькими входами можно передать кортеж значений или использовать вариант с передачей через state (см. полную реализацию в тестах).
9.6. Запекание подграфов
Механизм bakeSubgraph (см. core/baking.hpp) позволяет заменить целый подграф на узел AssetNode, загружающий предварительно вычисленный ассет. Это мощная оптимизация для редко изменяемых, но вычислительно затратных частей графа.
В тестах было показано, что запекание октавы шума снижает время выполнения с 12 мс до 0 мс при повторном запуске (загрузка из файла). Для сложных процедурных генераторов такая техника даёт значительный выигрыш.
9.7. Пример из реальной практики: генератор скалистой текстуры
В тестовом приложении реализован генератор текстуры скалы, использующий все описанные возможности.
Структура графа
Верхний уровень — RockTextureNode.
Создаёт локальный граф, содержащий по одному OctaveNode на каждую октаву.
Соединяет выходы октав с узлом BlendTexturesNode, который смешивает их с весами.
Уровень октавы — OctaveNode.
Строит ещё один локальный граф: NoiseNode → BlurNode.
BlurNode внутри использует функциональный компонент blurTexture для быстрой обработки пикселей.
Кэширование:
Повторный запуск всего графа (без изменения параметров) использует встроенное кэширование INode, сокращая время с 21 мс до 10 мс.
Отдельные октавы могут быть запечены в файлы PNG с помощью BakedOctaveNode, что полностью исключает вычисления при следующих запусках.

9.8. Рекомендации по использованию
Используйте INode для:
Крупных блоков, требующих кэширования.
Узлов с метаданными (для визуального редактора).
Частей графа, которые планируется запекать или сериализовать.
Используйте Component для:
Лёгкой внутренней логики узлов.
Быстрого прототипирования алгоритмов.
Операций, которые не нужно кэшировать отдельно.
Создавайте локальные графы, когда:
Нужно изолировать выполнение (например, рекурсивная генерация).
Требуется переиспользовать существующие узлы внутри другого узла.
Выполняется процедура, которая может быть запечена в будущем.
Запекайте подграфы, если:
Результат редко меняется.
Вычисления занимают значительное время.
Требуется ускорить итеративную разработку.
9.9. Заключение
Архитектура 3tone предоставляет уникальную гибкость за счёт сочетания объектно-ориентированных узлов и функциональных компонентов. Вложенность графов не ограничена, а контексты позволяют безопасно изолировать подвычисления или, наоборот, организовать сквозную передачу данных. Эти возможности делают движок мощным инструментом для построения процедурных генераторов, систем визуального программирования и других приложений, где требуется гибкая композиция вычислительных блоков.

Дополнительные примеры кода можно найти в файле mainlev2.cpp (практические тесты) и mainlevels.cpp (исследовательские тесты) в репозитории проекта.
